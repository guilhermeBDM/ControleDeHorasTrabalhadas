	create table tb_usuario(
	id_usuario int primary key auto_increment not null,

	create view vw_pedidos as
	Select tb_pedido.id_pedido as id_pedido,nm_pedido as nm_pedido,vl_produto as vl_produto,nr_quantidade as nr_quantidade,nm_fantasia as nm_fantasia
    from tb_pedido_item inner join tb_pedido 
    on tb_pedido_item.id_pedido = tb_pedido.id_pedido 
    inner join tb_fornecedor on tb_pedido.id_fornecedor = tb_fornecedor.id_fornecedor;

    create view vw_consulta_cinema as
	SELECT tb_ingresso_sessao.id_ingresso_sessao, ds_cpf,nr_cadeira,dt_compra,vl_preco,nm_filme,bt_meia from tb_ingresso_sessao INNER JOIN tb_sessao on tb_ingresso_sessao.id_sessao = tb_sessao.id_sessao 
    INNER JOIN tb_preco_sessao on tb_ingresso_sessao.id_preco_sessao = tb_preco_sessao.id_preco_sessao;
             
	create view vw_consulta_fliperama as
	SELECT tb_ingresso_fliperama.id_ingresso_fliperama, ds_cpf,dt_compra,vl_preco,nr_fixas from tb_ingresso_fliperama inner join tb_preco_fliperama on tb_ingresso_fliperama.id_preco_fliperama = tb_preco_fliperama.id_preco_fliperama;
	        

	Create view vw_fluxo_caixa as
	SELECT date(dt_pedido) as dt_dia, sum(vl_produto * nr_quantidade) as vl_valor, 'Saída' as op_operacao, 'Pedidos' as ds_motivo
	from tb_pedido inner join tb_pedido_item 
	on tb_pedido.id_pedido = tb_pedido_item.id_pedido 
	group by dt_dia,op_operacao,ds_motivo
    UNION
    select date(dt_data_registro_folha) as dt_dia, sum(tb_brutos.vl_salariobruto) as vl_valor,'Saída' as op_operacao,'Pagamentos' as ds_motivo
    from tb_folha_de_pagamento inner join tb_brutos on tb_folha_de_pagamento.id_folha = tb_brutos.id_brutos
	group by dt_dia,op_operacao,ds_motivo
	UNION
	SELECT date(dt_compra) as dt_dia, 
	sum(CASE WHEN tb_ingresso_sessao.bt_meia = true THEN ROUND((vl_preco /2),1) ELSE vl_preco END ) as vl_valor, 
	'Entrada'as op_operacao,'Vendas' as ds_motivo
	from tb_ingresso_sessao inner join tb_preco_sessao on tb_ingresso_sessao.id_preco_sessao = tb_preco_sessao.id_preco_sessao 
    group by dt_dia,op_operacao,ds_motivo
    UNION
	SELECT date(dt_compra) as dt_dia, Round(sum(vl_preco * nr_fixas),1) as vl_valor, 
    'Entrada' as op_operacao,'Vendas' as ds_motivo
    from tb_ingresso_fliperama inner join tb_preco_fliperama on tb_ingresso_fliperama.id_preco_fliperama = tb_preco_fliperama.id_preco_fliperama
    group by dt_dia,op_operacao,ds_motivo
    UNION
	Select date(dt_data_da_transacao) as dt_dia,sum(vl_valor) as vl_valor,op_operacao,ds_motivo
    from tb_contabilidade
    group by dt_dia,op_operacao,ds_motivo;
    

    create view vw_consulta_solicitacoes as
    SELECT tb_alteracao_ponto.id_alteracao_ponto, tb_ponto.id_ponto, tb_usuario.nm_usuario, tb_ponto.dt_movimento ,
    tb_alteracao_ponto.dt_novo_movimento as 'dt_novomovimento',tb_ponto.ds_status , tb_alteracao_ponto.ds_status as 'ds_novostatus', 
    tb_alteracao_ponto.ds_motivo
    from ((tb_alteracao_ponto INNER JOIN tb_ponto on tb_alteracao_ponto.id_ponto = tb_ponto.id_ponto)left join tb_usuario on tb_usuario.id_usuario = tb_ponto.id_usuario);


	create view tb_chat_select as
	select max(tb_chat.id_chat) as id_chat, 
    tb_chat.bt_visualizou, 
    tb_chat.id_destinatario, 
    tb_chat.id_remetente,
    (select nm_usuario from tb_usuario where id_usuario = tb_chat.id_remetente limit 1) as nm_usuario
	from tb_usuario 
	left join tb_chat 
	on tb_usuario.id_usuario = tb_chat.id_destinatario
    where tb_chat.bt_visualizou = false
    group 
    by tb_chat.id_remetente
    union
	select 
    0 as id_chat, 
    true as bt_visualizou, 
    0 as id_destinatario, 
    id_usuario as id_remetente, 
    nm_usuario 
    from tb_usuario;
